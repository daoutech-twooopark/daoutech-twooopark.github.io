quickstart.sh
# 1 · 액세스 토큰 발급 (24h 유효)
curl -X POST https://api.bizppurio.com/v1/token \
  -H "Authorization: Basic {base64}"

# 2 · 메시지 발송
curl -X POST https://api.bizppurio.com/v3/message \
  -H "Authorization: Bearer {token}" \
  -d '{ "type":"sms", "from":"07000000000",
        "to":"01012345678",
        "content":{"sms":{"message":"..."}} }'
