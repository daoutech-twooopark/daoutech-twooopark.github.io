def send_with_retry(payload, max_retries=3):
    for attempt in range(max_retries):
        try:
            response = api.post('/v3/message', payload)
            if response.code == 1000:
                return response
            elif response.code in [3002, 3005]:
                refresh_token()
                continue
            elif response.code in [5002, 5003, 5004, 5005, 9000]:
                base = min(2 ** attempt * 100, 30000)  # ms
                jitter = random.uniform(0, base * 0.5)
                time.sleep((base + jitter) / 1000)
                continue
            else:
                return response   # 비재시도 오류
        except Exception:
            time.sleep(2 ** attempt)
    raise RuntimeError('재시도 한도 초과')
